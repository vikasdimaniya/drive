minio setup:
curl -O https://dl.min.io/server/minio/release/linux-amd64/minio
chmod +x minio
sudo mv minio /usr/local/bin/

minio start:
minio server --console-address ":9001" ~/minio-data

postgres setup:   
brew update   
brew install postgresql   
brew services start postgresql
